{% highlight ruby %} class PushNotificationService prepend SimpleCommand def initialize(user, payload) @user = user @payload = payload end def call subscription = @user.push_subscription return errors.add(:base, "No push subscription found") unless subscription Webpush.payload_send( message: JSON.generate(@payload), endpoint: subscription.endpoint, p256dh: subscription.p256dh_key, auth: subscription.auth_key, vapid: { subject: "mailto:#{ENV['VAPID_SUBJECT']}", public_key: ENV['VAPID_PUBLIC_KEY'], private_key: ENV['VAPID_PRIVATE_KEY'] } ) end end {% endhighlight %}
